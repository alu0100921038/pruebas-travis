def extraerHead ()
      extraer = @head
      headActual = @head.prev
      @head = headActual
      if headActual == nil
          @tail = nil
      else
          @head.next = nil
      end
      @size-=1
      return extraer
end

def insertarHead(value)
    nodo = Node.new(value)
    if @head == nil and @tail == nil
        @head = nodo
        @tail = nodo
    else
        @head.next = nodo
        nodo.prev = @head
        @head = nodo
    end
    @size+=1
end
